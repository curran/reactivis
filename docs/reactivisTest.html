function outputDataFlowGraph(name, model){
  model.detectFlowGraph(function (graph) {
    var json = JSON.stringify(graph, null, 2);
    fs.writeFile('./dataFlowGraphs/svg.json', json, function(err) {
      if(err) console.log(err);
    }); 
  });
}
